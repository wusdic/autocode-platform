# 多项目自动编程平台 · 从零开始操作手册

> 配套《01-最终设计方案.md》。本手册是可逐条执行的 step-by-step。
> 目标环境：一台 Ubuntu 22.04+ 服务器（≥8 核 / 32GB 内存 / 200GB 磁盘建议），有 sudo。

## 阶段 2 · 编写"项目启动器"脚本

这是整个平台的核心运维脚本。保存为 `~/platform/launch_project.sh`。

```bash
mkdir -p ~/platform
cat > ~/platform/launch_project.sh <<'LAUNCHER'
#!/usr/bin/env bash
set -euo pipefail

PROJECT_ID="${1:?usage: launch_project.sh <project_id> <base_port>}"
BASE_PORT="${2:?need base_port, e.g. 8650}"

PROOT="/data/projects/${PROJECT_ID}"
export HERMES_HOME="${PROOT}/.hermes"
WORKSPACE="${PROOT}/workspace"

mkdir -p "${WORKSPACE}/design" "${WORKSPACE}/src"

echo "==> [1/6] 创建独立 Hermes 实例目录 ${HERMES_HOME}"
# 用独立 HERMES_HOME 跑后续所有 hermes 命令（已 export，子命令自动继承）

echo "==> [2/6] 初始化 Kanban board（该项目专属）"
hermes kanban boards create "${PROJECT_ID}" \
  --name "${PROJECT_ID}" --description "auto-coding project ${PROJECT_ID}" --switch

echo "==> [3/6] 创建角色 profiles + 模型/描述"
# 不同角色配不同模型，实现多模型质疑。模型字符串按你的供应商替换。
create_role () {
  local name="$1" model="$2" desc="$3"
  hermes profile create "${name}" --description "${desc}" || true
  hermes -p "${name}" config set model.default "${model}"
  hermes -p "${name}" config set agent.max_turns 200
  hermes -p "${name}" config set terminal.cwd "${WORKSPACE}"
}

# CEO：只沟通（toolset 在第 4 步裁剪）
create_role ceo               "anthropic/claude-opus-4.6"        "Talks to user, splits requirements, routes tasks. Never codes."
# 产品委员会（不同模型）
create_role pm-lead           "anthropic/claude-sonnet-4.6"      "Orchestrates product council."
create_role pm-research-a     "openai/gpt-5.1"                   "Researches market, competitors, user scenarios."
create_role pm-research-b     "google/gemini-3-flash-preview"   "Hunts for counter-examples and risks."
create_role pm-critic         "anthropic/claude-opus-4.6"        "Challenges requirement gaps and over-design."
create_role pm-synthesizer    "anthropic/claude-sonnet-4.6"      "Synthesizes PRD."
# 架构委员会
create_role arch-lead         "anthropic/claude-sonnet-4.6"      "Orchestrates architecture council."
create_role arch-simple       "anthropic/claude-sonnet-4.6"      "Proposes lightest viable design."
create_role arch-scale        "openai/gpt-5.1"                   "Proposes scalable design."
create_role arch-security     "anthropic/claude-opus-4.6"        "Reviews security, isolation, permissions."
create_role arch-critic       "google/gemini-3-flash-preview"   "Finds coupling and ripple-effect risks."
create_role arch-synthesizer  "anthropic/claude-sonnet-4.6"      "Produces ADR + interface spec + TODO."
# 研发与质控
create_role dev-lead          "anthropic/claude-sonnet-4.6"      "Splits and links coding tasks. Does not code."
create_role dev-worker-1      "anthropic/claude-sonnet-4.6"      "Implements assigned task in its worktree only."
create_role dev-worker-2      "anthropic/claude-sonnet-4.6"      "Implements assigned task in its worktree only."
create_role qa                "anthropic/claude-opus-4.6"        "Writes/runs tests, blocks release, files defects."
create_role release           "anthropic/claude-sonnet-4.6"      "Merges/packages/deploys after QA gate."
create_role change-guardian   "anthropic/claude-opus-4.6"        "Change impact analysis and design gate."

echo "==> [4/6] 裁剪 toolset（第一层权限）+ 注入 SOUL/AGENTS"
# CEO：只能沟通+看板，无 terminal/file/patch
hermes -p ceo config set toolsets "clarify,kanban,memory,messaging"
cp ~/platform-base/templates/SOUL.ceo.md             "${HERMES_HOME}/profiles/ceo/SOUL.md" 2>/dev/null || true
cp ~/platform-base/templates/SOUL.change-guardian.md "${HERMES_HOME}/profiles/change-guardian/SOUL.md" 2>/dev/null || true
# 设计角色：看板+只写 design，无 terminal/patch
for r in pm-lead pm-critic arch-lead arch-critic change-guardian; do
  hermes -p "$r" config set toolsets "kanban,file,web,memory"
done
for r in pm-research-a pm-research-b arch-simple arch-scale arch-security pm-synthesizer arch-synthesizer; do
  hermes -p "$r" config set toolsets "web,file,memory"
done
# 研发总监：看板+读文件，不写业务码
hermes -p dev-lead config set toolsets "kanban,file,memory"
# 工程师/质控：Docker backend + worktree（真沙箱）
for r in dev-worker-1 dev-worker-2 qa release; do
  hermes -p "$r" config set toolsets "kanban,terminal,file,memory"
  hermes -p "$r" config set terminal.backend docker
  hermes -p "$r" config set terminal.docker_image "python:3.11-slim"
done
# 复制 base skills 快照到各 profile（只读模板）
for d in "${HERMES_HOME}"/profiles/*/; do
  mkdir -p "${d}skills"
  cp -rn ~/platform-base/skills/* "${d}skills/" 2>/dev/null || true
done
# 注入全局 AGENTS.md 到 workspace（worker 在 workdir 内会被加载）
cp ~/platform-base/templates/AGENTS.md "${WORKSPACE}/AGENTS.md"

echo "==> [5/6] 安装 policy-plugin（第二层权限 hook）"
mkdir -p "${HERMES_HOME}/plugins/policy"
cp ~/platform/policy_plugin.py "${HERMES_HOME}/plugins/policy/plugin.py"
cat > "${HERMES_HOME}/plugins/policy/plugin.yaml" <<EOF
name: policy
description: Role permission and design-gate enforcement
EOF

echo "==> [6/6] 配置 Kanban + API server，启动 gateway（内嵌 dispatcher）"
hermes config set kanban.dispatch_in_gateway true
hermes config set kanban.max_in_progress 3
hermes config set kanban.failure_limit 2
# 该项目 CEO 的 API 端口（供你的网关转发）
cat >> "${HERMES_HOME}/profiles/ceo/.env" <<EOF
API_SERVER_ENABLED=true
API_SERVER_PORT=${BASE_PORT}
API_SERVER_KEY=$(openssl rand -hex 16)
EOF
hermes -p ceo gateway install   # systemd 托管，开机自启、崩溃自重启
hermes -p ceo gateway start

echo "✅ 项目 ${PROJECT_ID} 就绪。CEO API 端口=${BASE_PORT}，HERMES_HOME=${HERMES_HOME}"
echo "   API_SERVER_KEY 见 ${HERMES_HOME}/profiles/ceo/.env"
LAUNCHER
chmod +x ~/platform/launch_project.sh
```

> ⚠️ 模型字符串（`anthropic/claude-opus-4.6` 等）按你实际供应商和当时可用模型替换。运行 `hermes model` 可看可选列表。

## 阶段 3 · 编写权限闸门插件（第二层硬拦）

保存为 `~/platform/policy_plugin.py`。

```bash
cat > ~/platform/policy_plugin.py <<'PLUGIN'
"""角色权限 + 设计闸门：pre_tool_call 硬拦。"""
import json, os
from pathlib import Path

# 哪些角色禁止执行类工具（兜底，即使 toolset 漏配）
NO_EXEC_ROLES = {
    "ceo", "pm-lead", "pm-critic", "arch-lead", "arch-critic",
    "change-guardian", "dev-lead",
    "pm-research-a", "pm-research-b", "pm-synthesizer",
    "arch-simple", "arch-scale", "arch-security", "arch-synthesizer",
}
EXEC_TOOLS = {"terminal", "patch", "write_file"}

def _role() -> str:
    # profile 名 = HERMES_HOME 末段目录名
    return Path(os.environ.get("HERMES_HOME", "")).name or "unknown"

def _approved_designs(ws: str) -> set:
    f = Path(ws) / "design" / "approved_versions.txt"
    if f.exists():
        return {l.strip() for l in f.read_text().splitlines() if l.strip()}
    return set()

def enforce(tool_name, args, task_id, **kwargs):
    role = _role()

    # 闸门 1：非执行角色不准碰执行类工具
    if role in NO_EXEC_ROLES and tool_name in EXEC_TOOLS:
        return {"action": "block",
                "message": f"Role '{role}' is not allowed to call '{tool_name}'. Create a kanban task for an executor role instead."}

    # 闸门 2 & 3：工程师改代码前，校验 design_version 与 allowed_paths
    if role.startswith("dev-worker") and tool_name in {"patch", "write_file"}:
        ws = os.environ.get("TERMINAL_CWD") or os.getcwd()
        # 必须有已批准的设计版本
        if not _approved_designs(ws):
            return {"action": "block",
                    "message": "No approved design_version found. Code changes require an approved design first."}
        # 目标文件必须在本 task 的 allowed_paths 内
        target = (args or {}).get("path", "")
        allow_file = Path(ws) / "design" / f"allowed_paths.{task_id}.txt"
        if allow_file.exists():
            allowed = [l.strip() for l in allow_file.read_text().splitlines() if l.strip()]
            if target and not any(target.startswith(p) for p in allowed):
                return {"action": "block",
                        "message": f"File '{target}' is outside this task's allowed_paths. Do not modify files beyond your task scope."}
    return None

def register(ctx):
    ctx.register_hook("pre_tool_call", enforce)
PLUGIN
```

> 设计批准流程：change-guardian / arch-synthesizer 完成设计后，把版本号追加进 `workspace/design/approved_versions.txt`，并为每个编码任务写 `allowed_paths.<task_id>.txt`。这两个文件就是设计闸门的"钥匙"。

## 阶段 4 · 编写 FastAPI 控制平面（对外唯一入口）

保存为 `~/platform/control_plane.py`。这是最小可用骨架。

```bash
python3 -m venv ~/platform/venv
~/platform/venv/bin/pip install fastapi uvicorn httpx psycopg2-binary redis

cat > ~/platform/control_plane.py <<'CP'
import os, subprocess, json, httpx
from fastapi import FastAPI, HTTPException, Header
from fastapi.responses import StreamingResponse
from pydantic import BaseModel

app = FastAPI(title="Auto-Coding Control Plane")

# 简单内存注册表（生产改 Postgres）
PROJECTS = {}   # id -> {port, key, home}
NEXT_PORT = [8650]
GATEWAY_TOKEN = os.environ.get("PLATFORM_TOKEN", "change-me")

def auth(token: str):
    if token != GATEWAY_TOKEN:
        raise HTTPException(401, "bad token")

class CreateProject(BaseModel):
    project_id: str

@app.post("/api/projects")
def create_project(body: CreateProject, x_token: str = Header(None)):
    auth(x_token)
    pid, port = body.project_id, NEXT_PORT[0]; NEXT_PORT[0] += 1
    subprocess.run(["bash", os.path.expanduser("~/platform/launch_project.sh"),
                    pid, str(port)], check=True)
    home = f"/data/projects/{pid}/.hermes"
    key = ""
    with open(f"{home}/profiles/ceo/.env") as f:
        for line in f:
            if line.startswith("API_SERVER_KEY="):
                key = line.split("=",1)[1].strip()
    PROJECTS[pid] = {"port": port, "key": key, "home": home}
    return {"project_id": pid, "port": port, "status": "ready"}

class Msg(BaseModel):
    message: str
    session_id: str = "main"

@app.post("/api/projects/{pid}/messages")
async def message_ceo(pid: str, body: Msg, x_token: str = Header(None)):
    auth(x_token)
    p = PROJECTS.get(pid) or HTTPException(404)
    url = f"http://127.0.0.1:{p['port']}/v1/chat/completions"
    headers = {"Authorization": f"Bearer {p['key']}",
               "X-Hermes-Session-Id": body.session_id,
               "X-Hermes-Session-Key": f"agent:ceo:webui:{body.session_id}"}
    payload = {"model": "ceo", "messages": [{"role": "user", "content": body.message}]}
    async with httpx.AsyncClient(timeout=600) as c:
        r = await c.post(url, headers=headers, json=payload)
        return r.json()

@app.get("/api/projects/{pid}/tasks")
async def list_tasks(pid: str, x_token: str = Header(None)):
    auth(x_token)
    p = PROJECTS.get(pid) or HTTPException(404)
    env = dict(os.environ, HERMES_HOME=p["home"])
    out = subprocess.run(["hermes", "kanban", "--board", pid, "list", "--json"],
                         env=env, capture_output=True, text=True)
    return json.loads(out.stdout or "[]")

@app.get("/api/projects/{pid}/requirements")
def requirements(pid: str, x_token: str = Header(None)):
    auth(x_token)
    f = f"/data/projects/{pid}/workspace/design/requirements.yaml"
    return {"requirements": open(f).read()} if os.path.exists(f) else {"requirements": None}
CP
```

启动控制平面：

```bash
PLATFORM_TOKEN="$(openssl rand -hex 16)" \
~/platform/venv/bin/uvicorn control_plane:app \
  --app-dir ~/platform --host 127.0.0.1 --port 9000
```

> 生产环境：把它装成 systemd 服务，前面架 nginx/Caddy 做 TLS + 对外鉴权。**绝不把 Hermes 的 8650+ 端口或 dashboard 暴露到公网**，只暴露 9000 这个网关。

## 阶段 9 · 部署 Watchdog（处理 90 轮/崩溃/超时）

保存为 `~/platform/watchdog.sh`，用 cron 每分钟跑。

```bash
cat > ~/platform/watchdog.sh <<'WD'
#!/usr/bin/env bash
set -euo pipefail
for proj_dir in /data/projects/*/; do
  pid=$(basename "$proj_dir")
  export HERMES_HOME="${proj_dir}.hermes"
  # 找出失败/超时/卡死的任务
  hermes kanban --board "$pid" list --status blocked --json 2>/dev/null \
   | jq -r '.[] | select(.last_event=="gave_up" or .last_event=="timed_out") | .id' \
   | while read -r tid; do
       [ -z "$tid" ] && continue
       # 自动建 continuation 卡，不等用户
       hermes kanban --board "$pid" create "Continue task ${tid}" \
         --assignee dev-worker-1 --goal --goal-max-turns 30 || true
       hermes kanban --board "$pid" comment "$tid" "watchdog: spawned continuation" || true
     done
done
WD
chmod +x ~/platform/watchdog.sh

# 加入 crontab，每分钟一次
( crontab -l 2>/dev/null; echo "* * * * * ~/platform/watchdog.sh >> ~/platform/watchdog.log 2>&1" ) | crontab -
```

## 常见故障速查

| 现象 | 排查 |
|---|---|
| worker 用错模型 | `hermes -p <role> config get model.default`；实测 `hermes -p <role> -z "report your model"` |
| 任务一直 ready 不跑 | gateway 是否在跑：`hermes -p ceo gateway status`；dispatcher 内嵌于 gateway |
| 代码产物消失 | 用了 scratch 工作区；改 `--workspace worktree:` |
| worker 被反复回收 | 长任务没发 heartbeat；确认 worker 每小时 kanban_heartbeat |
| hook 没生效 | `hermes -p <role> --accept-hooks ...` 或设 `HERMES_ACCEPT_HOOKS=1`；非 TTY 必须显式接受 |
| 改到范围外文件 | 确认 Docker backend + allowed_paths 文件存在且路径正确 |
| API 调用 401 | 检查 ceo/.env 的 API_SERVER_KEY 与网关转发 header 一致 |
